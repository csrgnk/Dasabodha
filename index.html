let allPosts = []; let currentDashakam = 1; let searchTimeout = null; document.addEventListener("DOMContentLoaded", () => { // Fetches the post list. Added timestamp to avoid cache issues. fetch("./data/posts.json?v=" + new Date().getTime()) .then(res => res.json()) .then(data => { allPosts = data; renderPage(); setupDashNav(); }) .catch(err => { console.error("Error loading posts:", err); document.getElementById("postsContainer").innerHTML = "Error loading content."; }); }); /** * Handles Clean URL Redirection * Instead of .html, it opens the URL like a folder/slug */ function openPost(url){ const cleanUrl = url.replace(".html", ""); window.location.href = window.location.origin + "/" + cleanUrl + "/"; } /** * Renders the list of Samasams based on the selected Dashakam */ function renderPage(searchTerm = "") { const container = document.getElementById("postsContainer"); let filtered = searchTerm ? allPosts.filter(p => p.title.toLowerCase().includes(searchTerm.toLowerCase())) : allPosts.filter(p => p.dashakam === currentDashakam) .sort((a, b) => { // Sorts by samasam number extracted from the URL const numA = parseInt(a.url.match(/samasam-(\d+)/)?.[1] || 0); const numB = parseInt(b.url.match(/samasam-(\d+)/)?.[1] || 0); return numA - numB; }); if (filtered.length === 0) { container.innerHTML = `
🙏 No Samasams found in this Dashakam yet.
`; return; } container.innerHTML = ""; filtered.forEach(post => { const card = document.createElement("article"); card.className = "post-card"; card.onclick = () => openPost(post.url); card.innerHTML = `

${post.title}

${post.description ? post.description.substring(0,120) : "Dasabodha Telugu Lyrics..."}...

`; container.appendChild(card); }); } /** * Creates 20 Dashakam selection buttons */ function setupDashNav(){ const wrap = document.getElementById("dashButtons"); wrap.innerHTML = ""; for(let i=1; i<=20; i++){ const btn = document.createElement("button"); btn.innerText = "Dashakam " + i; btn.className = "dash-btn" + (i === currentDashakam ? " active" : ""); btn.onclick = () => { currentDashakam = i; document.querySelectorAll(".dash-btn").forEach(b => b.classList.remove("active")); btn.classList.add("active"); document.getElementById("postSearch").value = ""; renderPage(); window.scrollTo({top: 0, behavior: 'smooth'}); }; wrap.appendChild(btn); } }
